        function verifyAndCorrectPropertyValues(obj)
 371 
            [~, ~, props, constraints] = iGetSettingsProps(obj);
 372 
            
 373 
            % Ensure these cell arrays are all column-oriented
 374 
            props = props(:);
 375

 377 
            values = hGetProperty(obj, props);

            values = values(:);
 379 
            
 380 
            % only bother with the non-default values

 382 
            
 383 
            correctedValues = values;
 384 
            % Validate all properties against their constraints in a batch
 385

 386

 387

 388 
                'UniformOutput', false);
 389

 391 
            changed = cellfun(@(x, y) ~isequal(x, y), values, correctedValues);
